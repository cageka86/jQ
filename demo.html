It could be usefull with infinite-ajax-scroll (copying all event handlers from a pattern item) (https://github.com/webcreate/infinite-ajax-scroll)
below example of initiation: 
            jQuery.ias({
                container : '.items',
                item: '.item',
                pagination: '#content .paginator',
                next: '.pagination_next',
                loader: '< img src="imgsrc"/>',
                triggerPageThreshold: 4,
                trigger : 'Show other',
                thresholdMargin : -100,
                history : false,
                loaderDelay : 10,
                onRenderComplete: function(items) {
                  $.each(items, function(i, v){
                    $(items[i]).cloneEvents('.good:eq(1)');
                  });
                }
            });
